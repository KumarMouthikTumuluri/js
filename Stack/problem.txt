let Stack = require('../stack')

function balancePar(expr){
  let pFlag = false;
  let nFlag = 0;
  let s = new Stack();
  for( let i of expr.split('') ){ 
    if( i === "(" ){
     	pFlag = true;
        }
  else if( pFlag && !Number.isNaN(Number.parseInt(i)) ){
        
        nFlag++;
        console.log('i->'+i+' '+'pFlag->'+pFlag+' '+'nFlag->'+nFlag); }
  else if( nFlag > 0 ) {
        i == ")" ? null : s.push(')');
        pFlag = false;
        nFlag = 0;
        console.log('i->'+i+' '+'pFlag->'+pFlag+' '+'nFlag->'+nFlag);
		}
  else if( i === ")"){continue;}
  s.push(i);
    
    
  }
  return s.data.join('');
}

function checkBalancedExpression (e) {
  const s = stack()
  let balanced = true

  for (const key in e) {
    if (e[key] === '(') {
      s.push(e[key])
    } else if (e[key] === ')') {
      if (s.length() === 0) {
        balanced = false
      }
      s.pop()
    }
  }
  if (s.length() > 0) {
      balanced = false
  }
  return balanced
